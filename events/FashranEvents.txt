country_event = { #Nozumol
    id = fashros.2
    title = fashros.2.t
    desc = fashros.2.d
    picture = MONGOL_BANNERS_eventPicture

    fire_only_once = yes

    mean_time_to_happen = {
		months = 30
	}

    trigger = {
        AND = {
            nuzoto_area = {
                type = all
                AND = {
                    owned_by = ROOT
                    religion = tonmo
                }
            }
            toza_area = {
                type = all
                owned_by = ROOT
                religion = tonmo
            }
        }
    }

    option = { #Change cultures
        name = fashros.2.a
        ai_chance = {
            factor = 10
        }
        custom_tooltip = nozumol_culture_tooltip
        hidden_effect = {
            5840 = {
                change_culture = Fisbiklan
                add_core = FSB
            }
            5841 = {
                change_culture = Fisbiklan
                add_core = FSB
            }
            5842 = {
                change_culture = Fisbiklan
                add_core = FSB
            }
            5846 = {
                change_culture = Jimbalan
                add_core = JIB
            }
            5844 = {
                change_culture = Central_Nhoman
                add_core = LAM
            }
            5845 = {
                change_culture = Central_Nhoman
                add_core = LAM
            }
        }
    }

    option = { #Accept culture, gain bonus
        name = fashros.2.b
        ai_chance = {
            factor = 1
        }
        add_accepted_culture = Nozuman
        every_province = {
            limit = {
                OR = {
                    area = nuzoto_area
                    area = toza_area
                }
            }
            add_province_modifier = {
                name = nozumol_culture_bonus
                duration = -1
            }
        }
    }
}
